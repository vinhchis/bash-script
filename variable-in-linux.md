[vinhchis@archlinux ~]$ echo $PS1
[\u@\h \W]\$
    \u: Tên người dùng

    \h: Tên máy chủ rút gọn (hostname)

    \w: Thư mục làm việc hiện tại (không có ~)

    \W: Thư mục làm việc hiện tại (với ~ nếu có)

    \$: Dấu nhắc lệnh ($ nếu bạn là người dùng bình thường, # nếu bạn là root)

    \t: Giờ hiện tại

    \d: Ngày hiện tại

    \a: Chuông (bell)
